            <h3 id="base-config" tabindex="-1">Base <code>eleventy.config.js</code> <a href="#base-config" aria-hidden="true" data-is-anchor="true">#</a></h3>
<p>The package includes a fully-configured Eleventy config file <code>eleventy.config.js</code> that you can symlink to your project to get:</p>
<ul>
<li>All eleventy-blades plugins enabled</li>
<li>Eleventy Navigation plugin</li>
<li>Table of Contents plugin (conditionally loaded if installed)</li>
<li>Markdown-it with anchors and attributes</li>
<li>YAML data support</li>
<li>CLI input directory support</li>
<li>Symlink support for development</li>
<li><em>and more</em></li>
</ul>
<p><strong>Benefits of symlinking:</strong></p>
<ul>
<li><strong>Always up-to-date</strong>: Configuration automatically updates when you upgrade the package</li>
<li><strong>Less maintenance</strong>: No need to manually sync configuration changes</li>
<li><strong>Quick setup</strong>: Get started immediately with best-practice configurations</li>
<li><strong>Easy customization</strong>: Override specific settings by creating your own config that imports from the symlinked version</li>
</ul>
<p><strong>Installation as simple as:</strong></p>
<pre><code class="language-sh">npm install @anyblades/eleventy-blades
ln -s ./node_modules/@anyblades/eleventy-blades/src/eleventy.config.js
</code></pre>
<details class="breakout-item"><summary role="button" class="outline">How it works</summary>
```js

import eleventyBladesPlugin from "@anyblades/eleventy-blades";

// markdown-it-anchor not installed
}
let markdownItAttrs;
try {
markdownItAttrs = (await import("markdown-it-attrs")).default;
} catch (e) {
// markdown-it-attrs not installed
}

import yaml from "js-yaml";

import fs from "node:fs";
import path from "node:path";</p>
<p>/**</p>
<ul>
<li>Eleventy Configuration</li>
<li>@param {Object} eleventyConfig - The Eleventy configuration object</li>
<li>@returns {Object} The Eleventy configuration object
*/
export default function (eleventyConfig) {
const inputDir = eleventyConfig.directories.input;</li>
</ul>
<p>/* Jekyll parity */
eleventyConfig.addPassthroughCopy("assets");
eleventyConfig.addGlobalData("layout", "default");
eleventyConfig.setLiquidOptions({
dynamicPartials: false, // allows unquoted Jekyll-style includes

eleventyConfig.directories.includes,
inputDir + "../_includes", // for shared multisite includes

<p>/* Plugins */
eleventyConfig.addPlugin(RenderPlugin);
if (eleventyNavigationPlugin) eleventyConfig.addPlugin(eleventyNavigationPlugin);
eleventyConfig.addPlugin(eleventyBladesPlugin, {
mdAutoNl2br: true,
mdAutoUncommentAttrs: true,
mdAutoRawTags: true,
autoLinkFavicons: true,
siteData: true,
filters: [
"attr_set",
"attr_includes",
"merge",
"remove_tag",
"if",

"fetch",
"section",

}
// <a href="https://www.11ty.dev/docs/plugins/rss/#virtual-template"><i><img src="https://www.google.com/s2/favicons?domain=www.11ty.dev&sz=64"></i> /docs/plugins/rss/#virtual-template</a>
if (feedPlugin) {

let siteData = {};

siteData = yaml.load(fs.readFileSync(<code>${inputDir}/_data/site.yml</code>, "utf8"));
} catch (e) {
// _data/site.yml not found
}
eleventyConfig.addPlugin(feedPlugin, {
type: "atom", // or "rss", "json"
outputPath: "/feed.xml",

name: "feed", // iterate over <code>collections.posts</code>

},

});
}</p>
<p>/* Libraries */
let md = markdownIt({

if (markdownItAnchor) {

            <h3 id="base-scripts" tabindex="-1">Base 11ty npm scripts <small>via npm workspace</small> <a href="#base-scripts" aria-hidden="true" data-is-anchor="true">#</a></h3>
<p>This package provides a pre-configured <code>do</code> folder setup that helps organize your development workflow using npm workspaces. The <code>do</code> folder contains scripts for building and running your Eleventy project.</p>
<p><strong>Installation:</strong></p>
<ol>
<li>Install <a href="https://github.com/anyblades/eleventy-blades"><i><img src="https://www.google.com/s2/favicons?domain=github.com&sz=64"></i> /anyblades/eleventy-blades</a> to reuse pre-defined 11ty scripts from there:</li>
</ol>
<pre><code class="language-sh">npm install @anyblades/eleventy-blades
</code></pre>
<ol start="2">
<li>Create a helper folder <code>do</code> to symlink the <code>do/package.json</code> within:</li>
</ol>
<pre><code class="language-sh">mkdir do
cd ./do
ln -s ../node_modules/@anyblades/eleventy-blades/src/do/package.json
</code></pre>
<ol start="3">
<li>Finally register <code>do</code> folder as npm workspace in your root <code>package.json</code>:</li>
</ol>
<pre><code data-caption="./package.json" class="language-json">{
  ...
  &quot;workspaces&quot;: [&quot;do&quot;],
  &quot;scripts&quot;: {
    &quot;start&quot;: &quot;npm -w do run start&quot;,
    &quot;stage&quot;: &quot;npm -w do run stage&quot;,
    &quot;build&quot;: &quot;npm -w do run build&quot;
  },
  ...
}
</code></pre>
<p><strong>Done!</strong> 🎉 Now you can run:</p>
<ul>
<li><code>npm start</code> to start 11ty dev server with live reload and Tailwind watch mode</li>
<li><code>npm run stage</code> to build and serve production-like site locally</li>
<li><code>npm run build</code> to finally build the site for production</li>
<li>all available scripts: <a href="https://github.com/anyblades/eleventy-blades/blob/main/src/do/package.json"><i><img src="https://www.google.com/s2/favicons?domain=github.com&sz=64"></i> /anyblades/eleventy-blades/blob/main/src/do/package.json</a></li>
</ul>
<p>Living examples:</p>
<ul>
<li><a href="https://github.com/anyblades/build-awesome-starter"><i><img src="https://www.google.com/s2/favicons?domain=github.com&sz=64"></i> /anyblades/build-awesome-starter</a></li>
<li><a href="https://github.com/anyblades/bladeswitch"><i><img src="https://www.google.com/s2/favicons?domain=github.com&sz=64"></i> /anyblades/bladeswitch</a></li>
</ul>
<p><strong>Benefits:</strong></p>
<ul>
<li><strong>Clean separation</strong>: Keep build scripts separate from project configuration</li>
<li><strong>Reusable workflows</strong>: Update scripts by upgrading the package</li>
<li><strong>Workspace isolation</strong>: Scripts run in their own workspace context</li>
<li><strong>Easy maintenance</strong>: No need to manually maintain build scripts</li>
</ul>

            <h3 id="data-helpers" tabindex="-1">Data helpers <a href="#data-helpers" aria-hidden="true" data-is-anchor="true">#</a></h3>
<p>Adds global <code>site</code> data to your Eleventy project, providing commonly needed values that can be accessed in all templates:</p>
<table>
<thead>
<tr>
<th>Variable</th>
<th>Value</th>
</tr>
</thead>
<tbody>
<tr>
<td><code>{{ site.year }}</code></td>
<td>The current year as a number (e.g., <code>2026</code>)</td>
</tr>
<tr>
<td><code>{{ site.prod }}</code></td>
<td>Boolean indicating if running in production mode (<code>true</code> for <code>eleventy build</code>, <code>false</code> for <code>eleventy serve</code>)</td>
</tr>
</tbody>
</table>
<details class="breakout-item"><summary role="button" class="outline">How it works</summary>
```js
/**
 * Add site.year and site.prod global data
 * - site.prod: Boolean indicating if running in production mode (build) vs development (serve)
 * - site.year: Sets the current year to be available in all templates as {{ site.year }}
 *
 * @param {Object} eleventyConfig - The Eleventy configuration object
 */
export function siteData(eleventyConfig) {
  eleventyConfig.addGlobalData("site.prod", () => process.env.ELEVENTY_RUN_MODE === "build");
  eleventyConfig.addGlobalData("site.year", () => new Date().getFullYear());
}
```
</details>
